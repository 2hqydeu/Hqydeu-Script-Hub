local KeybindStorageFile = "HqydeuLoader_Keybinds.json"
local SavedKeybinds = {}
local CanPersistKeybinds = type(isfile) == "function" and type(readfile) == "function" and type(writefile) == "function"

if CanPersistKeybinds then
	pcall(function()
		if isfile(KeybindStorageFile) then
			local decoded = HttpService:JSONDecode(readfile(KeybindStorageFile))
			if type(decoded) == "table" then SavedKeybinds = decoded end
		end
	end)
end

GuiSizeValue = math.clamp(tonumber(SavedKeybinds.GuiSize) or 50, 20, 100)
function GuiScaleFromValue(value)
	return InterfaceScale * (value / 50)
end

function LoadKeybind(featureName)
	local savedName = SavedKeybinds[featureName]
	if type(savedName) == "string" then
		local success, keyCode = pcall(function() return Enum.KeyCode[savedName] end)
		if success and keyCode then return keyCode end
	end
	return Enum.KeyCode.Unknown
end

function SavePersistentSettings()
	if CanPersistKeybinds then
		pcall(function() writefile(KeybindStorageFile, HttpService:JSONEncode(SavedKeybinds)) end)
	end
end

function SaveKeybind(featureName, keyCode)
	SavedKeybinds[featureName] = keyCode.Name
	SavePersistentSettings()
end

function LoadColor(settingName, fallback)
	local saved = SavedKeybinds[settingName]
	if type(saved) == "table" and tonumber(saved.R) and tonumber(saved.G) and tonumber(saved.B) then
		return Color3.fromRGB(saved.R, saved.G, saved.B)
	end
	return fallback
end

function SaveThemeColors()
	SavedKeybinds.ThemeFrame = nil
	SavedKeybinds.ThemeAccent = {
		R = math.floor(AccentColor.R * 255 + 0.5),
		G = math.floor(AccentColor.G * 255 + 0.5),
		B = math.floor(AccentColor.B * 255 + 0.5),
	}
	SavePersistentSettings()
end

function KeybindText(keyCode)
	return keyCode == Enum.KeyCode.Unknown and "..." or keyCode.Name
end

SafeFeatureSystem = {Initializers = {}, Errors = {}}

function SafeFeatureSystem.Run(label, callback)
	local ok, err = pcall(callback)
	if not ok then
		local message = tostring(label) .. ": " .. tostring(err)
		table.insert(SafeFeatureSystem.Errors, message)
		pcall(function() warn("[HqydeuLoader] " .. message) end)
	end
	return ok
end

function SafeFeatureSystem.Add(name, initializer)
	if type(initializer) == "function" then
		table.insert(SafeFeatureSystem.Initializers, {Name = tostring(name or "Feature"), Run = initializer})
	end
end

function SafeFeatureSystem.RunInitializers()
	for _, entry in ipairs(SafeFeatureSystem.Initializers) do
		SafeFeatureSystem.Run(entry.Name, entry.Run)
	end
end

ThemeObjects = { Backgrounds = {}, Texts = {}, Accents = {}, AccentBackgrounds = {}, StateTexts = {} }

ThemeObjects.ApplyMainGradient = function(intensity)
	intensity = math.clamp(intensity or 0.5, 0, 0.5)
	local bgColor = FrameThemeColor
	MainTextureGradient.Color = ColorSequence.new({
		ColorSequenceKeypoint.new(0, bgColor:Lerp(Color3.new(0, 0, 0), 0.08)),
		ColorSequenceKeypoint.new(0.18, bgColor:Lerp(AccentColor, intensity * 0.38)),
		ColorSequenceKeypoint.new(0.48, bgColor:Lerp(AccentColor, intensity)),
		ColorSequenceKeypoint.new(0.73, bgColor:Lerp(AccentColor, intensity * 0.72)),
		ColorSequenceKeypoint.new(1, bgColor:Lerp(AccentColor, intensity * 0.22)),
	})
	if TargetTextureGradient then TargetTextureGradient.Color = MainTextureGradient.Color end
	if ThemeObjects.PlayerTP and ThemeObjects.PlayerTP.Gradient then ThemeObjects.PlayerTP.Gradient.Color = MainTextureGradient.Color end
end

function IsFrameLight()
	return (FrameThemeColor.R * 0.299 + FrameThemeColor.G * 0.587 + FrameThemeColor.B * 0.114) > 0.58
end

function InactiveTextColor()
	return IsFrameLight() and Color3.fromRGB(90, 90, 90) or Color3.fromRGB(180, 180, 180)
end

ThemeObjects.StyleTextOutline = function(obj)
	if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
		obj.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
		obj.TextStrokeTransparency = obj == SplashText and 1 or 0.5
	end
end

ThemeObjects.RefreshToggleButton = function(obj)
	if not obj:IsA("TextButton") or (obj.Text ~= "On" and obj.Text ~= "Off") then return end
	obj.BackgroundColor3 = obj.Text == "On" and AccentColor or FrameThemeColor:Lerp(Color3.new(0, 0, 0), 0.22)
	obj.TextColor3 = Color3.fromRGB(255, 255, 255)
end

ThemeObjects.StyleControlOutline = function(obj)
	if not obj:IsA("GuiObject") then return end
	if obj:FindFirstChild("ControlOutline") then return end
	local isVisibleButton = obj:IsA("TextButton") and obj.BackgroundTransparency < 0.9
	local isTile = obj:IsA("Frame") and (obj.Name:find("Card", 1, true) or obj.Name:find("Row", 1, true))
	if not isVisibleButton and not isTile then return end
	local outline = Instance.new("UIStroke")
	outline.Name = "ControlOutline"
	outline.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	outline.Color = Color3.fromRGB(0, 0, 0)
	outline.Thickness = 1
	outline.Transparency = 0.18
	outline.Parent = obj
end

function CompactGuiText(root)
	for _, obj in ipairs(root:GetDescendants()) do
		ThemeObjects.StyleTextOutline(obj)
		ThemeObjects.StyleControlOutline(obj)
		local isMainContent = obj:IsDescendantOf(MainFrame)
		if isMainContent and obj:IsA("TextButton") and (obj.Name == "MM2GameButton" or obj.Name == "QuietOrDieGameButton") then
			obj.Font = Enum.Font.GothamBold
			obj.TextSize = 11
			obj.TextWrapped = false
			obj.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
			obj.TextStrokeTransparency = 0.5
		elseif isMainContent and (obj:IsA("TextButton") or obj:IsA("TextBox")) then
			obj.TextSize = math.min(obj.TextSize, 12)
			obj.TextWrapped = true
		elseif isMainContent and obj:IsA("TextLabel") and obj.TextSize > 13 then
			obj.TextSize = 13
		end
	end
end

function UpdateTheme(instant)
	local bgColor = FrameThemeColor
	local lightTheme = IsFrameLight()
	local secBgColor = bgColor:Lerp(Color3.new(0, 0, 0), 0.22)
	local textColor = lightTheme and Color3.fromRGB(45, 45, 45) or Color3.fromRGB(240, 240, 240)
	local subTextColor = InactiveTextColor()
	local tweenInfo = TweenInfo.new(instant and 0 or 0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
	
	TweenService:Create(MainFrame, tweenInfo, {BackgroundColor3 = bgColor}):Play()
	if SplashFrame and SplashFrame.Parent then SplashFrame.BackgroundColor3 = bgColor end
	ThemeObjects.ApplyMainGradient(0.5)
	MainTextureStroke.Color = AccentColor
	if ThemeObjects.TabScroll and ThemeObjects.TabScroll.Parent then ThemeObjects.TabScroll.ScrollBarImageColor3 = AccentColor end
	for _, line in ipairs(MainPatternLines) do
		if line.Parent then line.BackgroundColor3 = AccentColor end
	end
	for i = #ThemeObjects.Backgrounds, 1, -1 do
		local obj = ThemeObjects.Backgrounds[i]
		if obj.Parent then
			if (obj:IsA("TextButton") or obj:IsA("TextBox")) then
				local keepSolid = obj.Name == "DestroyBtn" or obj.Name == "ResetCharacterBtn" or obj.Name == "RejoinBtn"
					or obj.Name == "ServerHopBtn" or obj.Name == "GuiSizeBtn" or obj.Name == "ClearKeybindsBtn"
				obj.BackgroundTransparency = keepSolid and 0 or 0.8
			end
		local target = (obj:IsA("TextButton") or obj:IsA("TextBox") or obj.Name == "TabScroll" or obj.Name == "TargetRow") and secBgColor or bgColor
			if obj.Name == "PlayerTPFrame" then target = Color3.fromRGB(14, 18, 26)
			end
			if obj:IsA("TextButton") and obj.Text == "On" then target = AccentColor end
			if obj.Name == "DestroyBtn" then target = Color3.fromRGB(125, 35, 40)
			elseif obj.Name == "ResetCharacterBtn" then target = Color3.fromRGB(125, 88, 18)
			elseif obj.Name == "RejoinBtn" then target = Color3.fromRGB(30, 135, 155)
			elseif obj.Name == "ServerHopBtn" then target = Color3.fromRGB(20, 45, 150)
			elseif obj.Name == "GuiSizeBtn" then target = Color3.fromRGB(35, 120, 35)
			elseif obj.Name == "ClearKeybindsBtn" then target = Color3.fromRGB(100, 35, 130) end
			TweenService:Create(obj, tweenInfo, {BackgroundColor3 = target}):Play()
		else
			table.remove(ThemeObjects.Backgrounds, i)
		end
	end
	for i = #ThemeObjects.Texts, 1, -1 do
		local obj = ThemeObjects.Texts[i]
		if obj.Parent then
			local parentName = obj.Parent and obj.Parent.Name or ""
			local isTileTitle = obj:IsA("TextLabel") and (parentName:find("Card", 1, true) or parentName:find("Row", 1, true))
			local isToggleText = obj:IsA("TextButton") and (obj.Text == "On" or obj.Text == "Off")
			local target = (obj.Name == "PageHeader" or obj.Name == "NavTabButton" or isTileTitle or isToggleText) and Color3.fromRGB(255, 255, 255)
				or (obj.Parent and (obj.Parent:IsA("TextButton") or obj:IsA("TextBox")) and textColor or subTextColor)
			TweenService:Create(obj, tweenInfo, {TextColor3 = target}):Play()
		else
			table.remove(ThemeObjects.Texts, i)
		end
	end
	for i = #ThemeObjects.Accents, 1, -1 do
		local obj = ThemeObjects.Accents[i]
		if obj.Parent then
			if obj:IsA("UIStroke") then obj.Color = AccentColor
			elseif obj:IsA("GuiObject") then obj.TextColor3 = AccentColor end
		else
			table.remove(ThemeObjects.Accents, i)
		end
	end
	for i = #ThemeObjects.AccentBackgrounds, 1, -1 do
		local obj = ThemeObjects.AccentBackgrounds[i]
		if obj.Parent then
			TweenService:Create(obj, tweenInfo, {BackgroundColor3 = AccentColor}):Play()
			if obj:IsA("TextButton") then
				obj.TextColor3 = Color3.fromRGB(255, 255, 255)
			end
		else
			table.remove(ThemeObjects.AccentBackgrounds, i)
		end
	end
	for i = #ThemeObjects.StateTexts, 1, -1 do
		local entry = ThemeObjects.StateTexts[i]
		if entry.Object and entry.Object.Parent then
			local stateColor = Color3.fromRGB(255, 255, 255)
			TweenService:Create(entry.Object, tweenInfo, {TextColor3 = stateColor}):Play()
		else
			table.remove(ThemeObjects.StateTexts, i)
		end
	end
end

function ThemeObjects.ExtrasFeature.BuildPath(instance)
	if not instance then return "" end
	local nodes = {}
	local current = instance
	while current and current.Parent and current.Parent ~= game do
		table.insert(nodes, 1, current)
		current = current.Parent
	end
	if not current or current.Parent ~= game then return instance:GetFullName() end
	local function escapeName(name) return tostring(name):gsub("\\", "\\\\"):gsub('"', '\\"') end
	local path = 'game:GetService("' .. escapeName(current.Name) .. '")'
	for _, node in ipairs(nodes) do
		if node.Name:match("^[%a_][%w_]*$") then path = path .. "." .. node.Name
		else path = path .. '["' .. escapeName(node.Name) .. '"]' end
	end
	return path
end

function RunGameModule(moduleName)
	local existing = LoadedGameModules[moduleName]
	if type(existing) == "table" and type(existing.Open) == "function" then
		existing.Open()
		return existing
	end
	local url = GameModuleUrls[moduleName]
	if not url or url:find("PASTE_", 1, true) then
		warn("Missing GitHub raw URL for " .. tostring(moduleName))
		return nil
	end
	local okSource, source = pcall(function() return game:HttpGet(url) end)
	if not okSource then warn("Failed to fetch " .. tostring(moduleName) .. ": " .. tostring(source)) return nil end
	local runner, compileError = loadstring(source)
	if not runner then warn("Failed to compile " .. tostring(moduleName) .. ": " .. tostring(compileError)) return nil end
	local okFactory, factory = pcall(runner)
	if not okFactory then warn("Failed to start " .. tostring(moduleName) .. ": " .. tostring(factory)) return nil end
	if type(factory) ~= "function" then warn(tostring(moduleName) .. " must return function(ctx)") return nil end
	local okModule, moduleApi = pcall(factory, BuildGameContext())
	if not okModule then warn("Failed to run " .. tostring(moduleName) .. ": " .. tostring(moduleApi)) return nil end
	LoadedGameModules[moduleName] = moduleApi or true
	return LoadedGameModules[moduleName]
end

ThemeObjects.ResolveESPCharacter = function(targetPlayer)
	if not targetPlayer then return nil end
	-- Infinite Yield follows Player.Character first. QOD's PlayersFolder may contain
	-- an incomplete/hidden proxy while the local player is the killer.
	local character = targetPlayer.Character
	if character and character:IsA("Model") then return character end
	local playersFolder = workspace:FindFirstChild("PlayersFolder")
	local folderCharacter = playersFolder and playersFolder:FindFirstChild(targetPlayer.Name)
	if folderCharacter and folderCharacter:IsA("Model") then return folderCharacter end
	return nil
end

			ThemeObjects.ESPHighlights[targetPlayer]:Destroy()
			ThemeObjects.ESPHighlights[targetPlayer] = nil
		end
		local adornments = ThemeObjects.ESPAdornments[targetPlayer]
		if adornments then
			for _, adornment in pairs(adornments) do
				if adornment then adornment:Destroy() end
			end
			ThemeObjects.ESPAdornments[targetPlayer] = nil
		end
		return
	end
	for playerObject, highlight in pairs(ThemeObjects.ESPHighlights) do
		if highlight then highlight:Destroy() end
		ThemeObjects.ESPHighlights[playerObject] = nil
	end
	for playerObject, adornments in pairs(ThemeObjects.ESPAdornments) do
		for _, adornment in pairs(adornments) do
			if adornment then adornment:Destroy() end
		end
		ThemeObjects.ESPAdornments[playerObject] = nil
	end
end

function FlingPlayer(targetPlayer)
	if Flinging or not targetPlayer or targetPlayer == Player then return end
	local character = Player.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local root = character and character:FindFirstChild("HumanoidRootPart")
	local targetCharacter = targetPlayer.Character
	local targetHumanoid = targetCharacter and targetCharacter:FindFirstChildOfClass("Humanoid")
	local targetRoot = targetCharacter and targetCharacter:FindFirstChild("HumanoidRootPart")
	if not humanoid or humanoid.Health <= 0 or not root or not targetHumanoid or targetHumanoid.Health <= 0 or not targetRoot then return end

	Flinging = true
	local originalCFrame = root.CFrame
	local originalAutoRotate = humanoid.AutoRotate
	local originalRootCollision = root.CanCollide
	humanoid.AutoRotate = false
	root.CanCollide = true

	local angularVelocity = Instance.new("BodyAngularVelocity")
	angularVelocity.Name = "R15FlingVelocity"
	angularVelocity.AngularVelocity = Vector3.new(0, 850000, 0)
	angularVelocity.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
	angularVelocity.P = math.huge
	angularVelocity.Parent = root

	pcall(function()
		local targetContactPart = targetCharacter:FindFirstChild("UpperTorso") or targetRoot
		local offsets = {
			CFrame.new(0, 0, 0),
			CFrame.new(0.7, 0, 0),
			CFrame.new(-0.7, 0, 0),
			CFrame.new(0, 0, 0.7),
			CFrame.new(0, 0, -0.7),
			CFrame.new(0, 1.1, 0),
			CFrame.new(0, -0.8, 0),
		}
		local startedAt = os.clock()
		local step = 0
		while os.clock() - startedAt < 1.8 and targetRoot.Parent and targetHumanoid.Health > 0 and root.Parent do
			step += 1
			local prediction = targetRoot.AssemblyLinearVelocity * 0.035
			local targetCFrame = targetContactPart.CFrame + prediction
			root.CFrame = targetCFrame * offsets[((step - 1) % #offsets) + 1]
			root.AssemblyAngularVelocity = Vector3.new(0, 850000, 0)
			root.AssemblyLinearVelocity = targetRoot.AssemblyLinearVelocity
			if os.clock() - startedAt > 0.45 and targetRoot.AssemblyLinearVelocity.Magnitude > 95 then break end
			RunService.Heartbeat:Wait()
		end
	end)

	if angularVelocity.Parent then angularVelocity:Destroy() end
	if root and root.Parent then
		root.CanCollide = originalRootCollision
		root.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
		root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
		root.CFrame = originalCFrame
	end
	if humanoid and humanoid.Parent then
		humanoid.AutoRotate = originalAutoRotate
	end
	Flinging = false
end

function ThemeObjects.WaypointUI.GetSegments(instance)
	local segments = {}
	local current = instance
	while current and current ~= game do
		table.insert(segments, 1, current.Name)
		current = current.Parent
	end
	return segments
end

function ThemeObjects.WaypointUI.Resolve(waypoint)
	local segments = waypoint.Segments
	if type(segments) ~= "table" or #segments == 0 then return nil end
	local current = game
	for _, name in ipairs(segments) do
		if current == game then
			local success, service = pcall(function() return game:GetService(name) end)
			current = success and service or game:FindFirstChild(name)
		else
			current = current and current:FindFirstChild(name)
		end
		if not current then return nil end
	end
	if current:IsA("BasePart") then return current end
	return nil
end

function ThemeObjects.WaypointUI.GetCFrame(waypoint)
	if type(waypoint.CFrame) == "table" and #waypoint.CFrame >= 12 then
		return CFrame.new(table.unpack(waypoint.CFrame, 1, 12))
	end
	local part = ThemeObjects.WaypointUI.Resolve(waypoint)
	return part and part.CFrame or nil
end

function ThemeObjects.WaypointUI.Save()
	local serializable = {}
	for _, waypoint in ipairs(ThemeObjects.WaypointUI.Waypoints) do
		table.insert(serializable, {Segments = waypoint.Segments, CFrame = waypoint.CFrame, Name = waypoint.Name, Kind = waypoint.Kind})
	end
	SavedKeybinds.WaypointsByPlace[ThemeObjects.WaypointUI.StorageKey] = serializable
	SavePersistentSettings()
end

-- Movement Logic
function SetupBinding(button, type)
	button.MouseButton1Click:Connect(function()
		PlayClick()
		button.Text = "..."
		if type == "Speed" then IsBindingSpeed = true
		elseif type == "Noclip" then IsBindingNoclip = true
		elseif type == "Fly" then IsBindingFly = true
		elseif type == "FreeCam" then ThemeObjects.FreeCam.Binding = true
		elseif type == "AutoLock" then IsBindingAutoLock = true
		elseif type == "ESP" then IsBindingESP = true end
		
		local conn
		conn = UserInputService.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.Keyboard then
				local selectedKey = input.KeyCode
				if selectedKey == Enum.KeyCode.Backspace or selectedKey == Enum.KeyCode.Delete then
					selectedKey = Enum.KeyCode.Unknown
				end
				if type == "Speed" then SpeedKeybind = selectedKey; IsBindingSpeed = false
				elseif type == "Noclip" then NoclipKeybind = selectedKey; IsBindingNoclip = false
				elseif type == "Fly" then FlyKeybind = selectedKey; IsBindingFly = false
				elseif type == "FreeCam" then ThemeObjects.FreeCam.Keybind = selectedKey; ThemeObjects.FreeCam.Binding = false
				elseif type == "AutoLock" then AutoLockKeybind = selectedKey; IsBindingAutoLock = false
				elseif type == "ESP" then ESPKeybind = selectedKey; IsBindingESP = false end
				
				SaveKeybind(type, selectedKey)
				button.Text = KeybindText(selectedKey)
				conn:Disconnect()
			end
		end)
	end)
end

function ThemeObjects.FreeCam.Enable()
	if ThemeObjects.FreeCam.Enabled then return end
	local camera = workspace.CurrentCamera
	if not camera then return end
	ThemeObjects.FreeCam.Enabled = true
	ThemeObjects.FreeCam.OriginalType = camera.CameraType
	ThemeObjects.FreeCam.OriginalSubject = camera.CameraSubject
	ThemeObjects.FreeCam.OriginalCFrame = camera.CFrame
	ThemeObjects.FreeCam.OriginalFocus = camera.Focus
	ThemeObjects.FreeCam.OriginalMouseBehavior = UserInputService.MouseBehavior
	ThemeObjects.FreeCam.OriginalMouseIcon = UserInputService.MouseIconEnabled
	ThemeObjects.FreeCam.Position = camera.CFrame.Position
	ThemeObjects.FreeCam.Pitch, ThemeObjects.FreeCam.Yaw = camera.CFrame:ToOrientation()
	camera.CameraType = Enum.CameraType.Scriptable
	local character = Player.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local root = character and character:FindFirstChild("HumanoidRootPart")
	if humanoid then humanoid:Move(Vector3.zero) end
	if root then root.AssemblyLinearVelocity = Vector3.zero root.AssemblyAngularVelocity = Vector3.zero end
	ThemeObjects.FreeCam.Toggle.Text = "On"
	ThemeObjects.FreeCam.Title.TextColor3 = AccentColor
	game:GetService("ContextActionService"):BindActionAtPriority(
		"HqydeuFreeCamSink",
		function() return Enum.ContextActionResult.Sink end,
		false,
		Enum.ContextActionPriority.High.Value + 100,
		Enum.KeyCode.W, Enum.KeyCode.A, Enum.KeyCode.S, Enum.KeyCode.D,
		Enum.KeyCode.Q, Enum.KeyCode.E, Enum.KeyCode.LeftShift
	)
end

function ThemeObjects.FreeCam.Disable()
	if not ThemeObjects.FreeCam.Enabled then return end
	ThemeObjects.FreeCam.Enabled = false
	game:GetService("ContextActionService"):UnbindAction("HqydeuFreeCamSink")
	-- Always release the RMB camera lock. Restoring a captured LockCenter state can
	-- leave the cursor pinned after Free Cam is turned off.
	UserInputService.MouseBehavior = Enum.MouseBehavior.Default
	UserInputService.MouseIconEnabled = true
	local camera = workspace.CurrentCamera
	if camera then
		camera.CameraType = ThemeObjects.FreeCam.OriginalType or Enum.CameraType.Custom
		local character = Player.Character
		local humanoid = character and character:FindFirstChildOfClass("Humanoid")
		camera.CameraSubject = humanoid or ThemeObjects.FreeCam.OriginalSubject
		if ThemeObjects.FreeCam.OriginalCFrame then camera.CFrame = ThemeObjects.FreeCam.OriginalCFrame end
		if ThemeObjects.FreeCam.OriginalFocus then camera.Focus = ThemeObjects.FreeCam.OriginalFocus end
	end
	ThemeObjects.FreeCam.Toggle.Text = "Off"
	ThemeObjects.FreeCam.Title.TextColor3 = InactiveTextColor()
	task.defer(function()
		-- Apply once more after Roblox's camera controller finishes this frame.
		if not ThemeObjects.FreeCam.Enabled then
			UserInputService.MouseBehavior = Enum.MouseBehavior.Default
			UserInputService.MouseIconEnabled = true
		end
	end)
end

function ThemeObjects.FreeCam.ToggleState()
	if ThemeObjects.FreeCam.Enabled then ThemeObjects.FreeCam.Disable() else ThemeObjects.FreeCam.Enable() end
end

ThemeObjects.FreeCam.Toggle.MouseButton1Click:Connect(function() PlayClick() ThemeObjects.FreeCam.ToggleState() end)
Connections.FreeCam = RunService.RenderStepped:Connect(function(deltaTime)
	if not ThemeObjects.FreeCam.Enabled then return end
	local camera = workspace.CurrentCamera
	if not camera then return end
	ThemeObjects.FreeCam.FocusedTextBox = nil
	pcall(function() ThemeObjects.FreeCam.FocusedTextBox = UserInputService:GetFocusedTextBox() end)
	if ThemeObjects.FreeCam.FocusedTextBox then
		UserInputService.MouseBehavior = Enum.MouseBehavior.Default
		return
	end
	if UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
		UserInputService.MouseBehavior = Enum.MouseBehavior.LockCurrentPosition
		local mouseDelta = UserInputService:GetMouseDelta()
		ThemeObjects.FreeCam.Yaw = ThemeObjects.FreeCam.Yaw - mouseDelta.X * 0.01
		ThemeObjects.FreeCam.Pitch = math.clamp(ThemeObjects.FreeCam.Pitch - mouseDelta.Y * 0.01, math.rad(-89), math.rad(89))
	else
		UserInputService.MouseBehavior = Enum.MouseBehavior.Default
	end
	local rotation = CFrame.fromOrientation(ThemeObjects.FreeCam.Pitch, ThemeObjects.FreeCam.Yaw, 0)
	local movement = Vector3.zero
	if UserInputService:IsKeyDown(Enum.KeyCode.W) then movement = movement + rotation.LookVector end
	if UserInputService:IsKeyDown(Enum.KeyCode.S) then movement = movement - rotation.LookVector end
	if UserInputService:IsKeyDown(Enum.KeyCode.D) then movement = movement + rotation.RightVector end
	if UserInputService:IsKeyDown(Enum.KeyCode.A) then movement = movement - rotation.RightVector end
	if UserInputService:IsKeyDown(Enum.KeyCode.E) then movement = movement + Vector3.new(0, 1, 0) end
	if UserInputService:IsKeyDown(Enum.KeyCode.Q) then movement = movement - Vector3.new(0, 1, 0) end
	local speed = UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) and 120 or 40
	if movement.Magnitude > 0 then ThemeObjects.FreeCam.Position = ThemeObjects.FreeCam.Position + movement.Unit * speed * deltaTime end
	local cameraCFrame = CFrame.new(ThemeObjects.FreeCam.Position) * rotation
	camera.CFrame = cameraCFrame
	camera.Focus = cameraCFrame * CFrame.new(0, 0, -512)
end)

function ToggleFly()
	FlyEnabled = not FlyEnabled
	FlyToggleBtn.Text = FlyEnabled and "On" or "Off"
	FlyTitle.TextColor3 = FlyEnabled and AccentColor or InactiveTextColor()
	local Character = Player.Character 
	local RootPart = Character and Character:FindFirstChild("HumanoidRootPart")
	if FlyEnabled then
		if RootPart then
			BodyVelocity = Instance.new("BodyVelocity") BodyVelocity.Velocity = Vector3.new(0, 0, 0) BodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge) BodyVelocity.Parent = RootPart
			BodyGyro = Instance.new("BodyGyro") BodyGyro.P = 9e4 BodyGyro.MaxTorque = Vector3.new(9e9, 9e9, 9e9) BodyGyro.CFrame = RootPart.CFrame BodyGyro.Parent = RootPart
		end
	else
		if BodyVelocity then BodyVelocity:Destroy() BodyVelocity = nil end
		if BodyGyro then BodyGyro:Destroy() BodyGyro = nil end
		if RootPart then RootPart.Velocity = Vector3.new(0, 0, 0) end
	end
end

function GetClosestPlayerToMouse()
	local nearest = nil
	local lastDist = math.huge
	local Camera = workspace.CurrentCamera
	local MousePos = UserInputService:GetMouseLocation()
	for _, p in pairs(Players:GetPlayers()) do
		if p ~= Player and p.Character and p.Character:FindFirstChild("Head") then
			local headPos, onScreen = Camera:WorldToViewportPoint(p.Character.Head.Position)
			if onScreen then
				local dist = (Vector2.new(headPos.X, headPos.Y) - MousePos).Magnitude
				if dist < lastDist then
					nearest = p
					lastDist = dist
				end
			end
		end
	end
	return nearest
end

Connections.MainLoop = RunService.RenderStepped:Connect(function()
	local Character = Player.Character 
    local RootPart = Character and Character:FindFirstChild("HumanoidRootPart") 
    local Humanoid = Character and Character:FindFirstChild("Humanoid") 
    local Camera = workspace.CurrentCamera

    if SpeedEnabled and RootPart and Humanoid then
		local MoveDir = Humanoid.MoveDirection 
        if MoveDir.Magnitude > 0 then 
            RootPart.CFrame = RootPart.CFrame + (MoveDir * (CurrentCFrameSpeed / 10)) 
        end
	end
    
	if FlyEnabled and RootPart and BodyGyro then
		local MoveDir = Vector3.new(0,0,0)
		if UserInputService:IsKeyDown(Enum.KeyCode.W) then MoveDir = MoveDir + Camera.CFrame.LookVector end
		if UserInputService:IsKeyDown(Enum.KeyCode.S) then MoveDir = MoveDir - Camera.CFrame.LookVector end
		if UserInputService:IsKeyDown(Enum.KeyCode.A) then MoveDir = MoveDir - Camera.CFrame.RightVector end
		if UserInputService:IsKeyDown(Enum.KeyCode.D) then MoveDir = MoveDir + Camera.CFrame.RightVector end
		if UserInputService:IsKeyDown(Enum.KeyCode.Space) then MoveDir = MoveDir + Vector3.new(0,1,0) end
		if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then MoveDir = MoveDir - Vector3.new(0,1,0) end
		if MoveDir.Magnitude > 0 then RootPart.CFrame = RootPart.CFrame + (MoveDir * (FlySpeed / 50)) end
		BodyGyro.CFrame = Camera.CFrame RootPart.Velocity = Vector3.new(0, 0, 0)
	end
	
	if AutoLockEnabled and UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) and RootPart then
		local target = GetClosestPlayerToMouse()
		if target and target.Character and target.Character:FindFirstChild("Head") then
			Camera.CFrame = CFrame.lookAt(Camera.CFrame.Position, target.Character.Head.Position)
		end
	end

	local refreshESPBody = os.clock() - (ThemeObjects.LastESPBodyRefresh or 0) >= 0.1
	if refreshESPBody then ThemeObjects.LastESPBodyRefresh = os.clock() end
    for _, p in pairs(Players:GetPlayers()) do
		local char = ThemeObjects.ResolveESPCharacter(p)
		local hum = char and char:FindFirstChildOfClass("Humanoid")
		local root = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso") or char.PrimaryPart)
        if p ~= Player and char and root and hum then

			if AntiFlingEnabled and not Flinging then
				local dangerous = root.AssemblyLinearVelocity.Magnitude > 90 or root.AssemblyAngularVelocity.Magnitude > 40
				for _, part in ipairs(char:GetDescendants()) do
					if part:IsA("BasePart") then
						if dangerous then
							if AntiFlingOriginalCollision[part] == nil then
								AntiFlingOriginalCollision[part] = part.CanCollide
							end
							part.CanCollide = false
						elseif AntiFlingOriginalCollision[part] ~= nil then
							part.CanCollide = AntiFlingOriginalCollision[part]
							AntiFlingOriginalCollision[part] = nil
						end
					end
				end
			end
            
			local isMurdererRoleESP = MurdererRoleESPEnabled and p == CurrentMurderer
			local isSheriffRoleESP = SheriffRoleESPEnabled and p == CurrentSheriff
			local roleColor = nil
			if isMurdererRoleESP then
				roleColor = Color3.fromRGB(255, 70, 70)
			elseif isSheriffRoleESP then
				roleColor = Color3.fromRGB(70, 130, 255)
			end
			local showPlayerESP = ESPEnabled or roleColor ~= nil

			-- Infinite Yield-style body ESP. These adornments live under CoreGui
			-- instead of the character, so local transparency/hide effects used by
			-- games such as QOD do not erase the overlay for the killer.
			if ThemeObjects.DeepESP.Enabled and roleColor == nil and refreshESPBody then
				local adornments = ThemeObjects.ESPAdornments[p]
				if not adornments then
					adornments = {}
					ThemeObjects.ESPAdornments[p] = adornments
				end
				for part, adornment in pairs(adornments) do
					if not part.Parent or not part:IsDescendantOf(char) then
						if adornment then adornment:Destroy() end
						adornments[part] = nil
					end
				end
				for _, part in ipairs(char:GetDescendants()) do
					if part:IsA("BasePart") then
						local adornment = adornments[part]
						if not adornment then
							adornment = Instance.new("BoxHandleAdornment")
							adornment.Name = "PlayerESPPart"
							adornment.Adornee = part
							adornment.AlwaysOnTop = true
							adornment.ZIndex = 10
							adornment.Parent = ThemeObjects.ESPAdornmentRoot
							adornments[part] = adornment
						end
						adornment.Size = part.Size
						adornment.Color3 = ESPColor
						adornment.Transparency = 0.64
					end
				end
			elseif not ThemeObjects.DeepESP.Enabled or roleColor ~= nil then
				ThemeObjects.ClearESPAdornments(p)
			end

			local oldCharacterHighlight = char:FindFirstChild("ESPHighlight")
			if oldCharacterHighlight then oldCharacterHighlight:Destroy() end
			local regularHighlight = ThemeObjects.ESPHighlights[p]
			if ESPEnabled and roleColor == nil then
				if not regularHighlight then
					regularHighlight = Instance.new("Highlight")
					regularHighlight.Name = "ESPHighlight"
					regularHighlight.Parent = ThemeObjects.ESPAdornmentRoot
					ThemeObjects.ESPHighlights[p] = regularHighlight
				end
				regularHighlight.Enabled = true
				regularHighlight.Adornee = char
				regularHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
				regularHighlight.FillColor = ESPColor
				regularHighlight.FillTransparency = 0.72
				regularHighlight.OutlineColor = ESPColor
				regularHighlight.OutlineTransparency = 0
			elseif regularHighlight then
				regularHighlight:Destroy()
				ThemeObjects.ESPHighlights[p] = nil
			end

			local head = char:FindFirstChild("Head")
			if showPlayerESP and head then
				local espGui = head:FindFirstChild("ESPGui")
				if not espGui then
					espGui = Instance.new("BillboardGui", head)
					espGui.Name = "ESPGui"
					espGui.Size = UDim2.new(0, 200, 0, 50)
					espGui.AlwaysOnTop = true
					espGui.StudsOffset = Vector3.new(0, 5, 0)
					local label = Instance.new("TextLabel", espGui)
					label.Name = "ESPLabel"
					label.Size = UDim2.new(1, 0, 1, 0)
					label.BackgroundTransparency = 1
					label.Font = Enum.Font.GothamBold
					label.TextSize = 12
					label.TextStrokeTransparency = 0
				end
				local dist = (RootPart and (RootPart.Position - root.Position).Magnitude) or 0
				espGui.ESPLabel.TextColor3 = roleColor or ESPColor
				espGui.ESPLabel.Text = string.format("%s\nHP: %d | %d Studs", p.DisplayName, math.floor(hum.Health), math.floor(dist))
			elseif head and head:FindFirstChild("ESPGui") then
				head.ESPGui:Destroy()
			end

        end
    end
end)

function Cleanup()
	if ThemeObjects.FreeCam and ThemeObjects.FreeCam.Enabled then ThemeObjects.FreeCam.Disable() end
	if ThemeObjects.ClearESPAdornments then ThemeObjects.ClearESPAdornments() end
	RestoreAntiFlingCollisions()
	RestoreKillHitboxes()
	GunDropESP.Clear()
	if ThemeObjects.CleanupPartSelector then ThemeObjects.CleanupPartSelector() end
	if ThemeObjects.FlushQuietHistory then ThemeObjects.FlushQuietHistory() end
	if ThemeObjects.RestoreFullBright then ThemeObjects.RestoreFullBright() end
	if ThemeObjects.RestoreQuietClutter then ThemeObjects.RestoreQuietClutter() end
	local Character = Player.Character
	if Character then
		local Hum = Character:FindFirstChild("Humanoid") local Root = Character:FindFirstChild("HumanoidRootPart")
		if Hum then Hum.WalkSpeed = 16 end
		if Root then 
			Root.Velocity = Vector3.new(0,0,0) 
			if Root:FindFirstChildOfClass("BodyVelocity") then Root:FindFirstChildOfClass("BodyVelocity"):Destroy() end
			if Root:FindFirstChildOfClass("BodyGyro") then Root:FindFirstChildOfClass("BodyGyro"):Destroy() end
		end
		for _, part in pairs(Character:GetDescendants()) do if part:IsA("BasePart") then part.CanCollide = true end end
	end
	for _, p in pairs(Players:GetPlayers()) do
		local cleanupCharacter = ThemeObjects.ResolveESPCharacter and ThemeObjects.ResolveESPCharacter(p) or p.Character
		if cleanupCharacter then
			if cleanupCharacter:FindFirstChild("ESPHighlight") then cleanupCharacter.ESPHighlight:Destroy() end
			if cleanupCharacter:FindFirstChild("MurdererRoleHighlight") then cleanupCharacter.MurdererRoleHighlight:Destroy() end
			if cleanupCharacter:FindFirstChild("SheriffRoleHighlight") then cleanupCharacter.SheriffRoleHighlight:Destroy() end
			if cleanupCharacter:FindFirstChild("Head") and cleanupCharacter.Head:FindFirstChild("ESPGui") then cleanupCharacter.Head.ESPGui:Destroy() end
		end
	end
	for _, connection in pairs(Connections) do if connection then connection:Disconnect() end end
	if CoinFarmConnection then CoinFarmConnection:Disconnect() end
	ScreenGui:Destroy()
end

-- FLY-IN TITLE SEQUENCE
task.spawn(function()
	SafeFeatureSystem.Run("Startup animation", function()
		SplashFrame.Rotation = -540
		local spinIn = TweenService:Create(SplashFrame, TweenInfo.new(0.55, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
			Size = UDim2.new(0, 200, 0, 200),
			Position = UDim2.new(0.5, 0, 0.5, 0),
			Rotation = 0,
		})
		spinIn:Play()
		spinIn.Completed:Wait()

		local revealTitle = TweenService:Create(SplashText, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			TextTransparency = 0,
			TextStrokeTransparency = 0.5,
		})
		revealTitle:Play()
		revealTitle.Completed:Wait()
		task.wait(0.5)

		local splashStroke = SplashFrame:FindFirstChildOfClass("UIStroke")
		if splashStroke then
			TweenService:Create(splashStroke, TweenInfo.new(0.4), {Transparency = 1}):Play()
		end
		TweenService:Create(SplashText, TweenInfo.new(0.25), {TextTransparency = 1, TextStrokeTransparency = 1}):Play()
		local spinOut = TweenService:Create(SplashFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
			Size = UDim2.new(0, 0, 0, 0),
			Rotation = 540,
			BackgroundTransparency = 1,
		})
		spinOut:Play()
		spinOut.Completed:Wait()
	end)
	if SplashFrame and SplashFrame.Parent then SplashFrame:Destroy() end
	MainFrame.Visible = true
end)
